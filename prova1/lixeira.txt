#include <iostream>
using namespace std;

int main()
{
    int numero = 0, i = 0, j = 0, p = 0;
    bool v = false;
    
    cin >> numero;

    while (i < numero)
    {
        while (j < i)
        {
            while (p < j)
            {
                if (i+j+p == numero) 
               {
                   v = true;
               }
               p++;
            }
            j++;
        }
        i++;
    }
    if (v)
    {
        cout << "SIM\n";
    }
    else
    {
        cout << "NAO\n";
    }
    
    return 0;
}
